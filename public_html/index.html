Python Code for Web Scraping


from bs4 import BeautifulSoup

# Load the HTML content
with open("universities.html", "r") as file:
    html_content = file.read()

soup = BeautifulSoup(html_content, "html.parser")

# Using find and find_all
first_university = soup.find("div", class_="university")
all_universities = soup.find_all("div", class_="university")

print("First university using find:")
print(first_university.h2.text)

print("\nAll universities using find_all:")
for uni in all_universities:
    print(uni.h2.text)

# Using select and select_one
first_university_select = soup.select_one(".university")
all_universities_select = soup.select(".university")

print("\nFirst university using select_one:")
print(first_university_select.h2.text)

print("\nAll universities using select:")
for uni in all_universities_select:
    print(uni.h2.text)
